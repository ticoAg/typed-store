# TypedStore v1.0 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship a release-ready `TypedStore` v1.0 by freezing the public API, adding first-class bulk mutation APIs, and aligning packaging, documentation, examples, and release workflows with the final public contract.

**Architecture:** Build `v1.0` on top of the existing protocol-first `v0.3` surface. Add explicit bulk capability protocols and store/model-view methods without reintroducing compatibility layers, then lock the stable export surface and release path around those contracts. Keep request objects as the single input model and enforce bulk-specific query restrictions through explicit contract errors rather than silent coercion.

**Tech Stack:** Python 3.12+, SQLAlchemy 2.x, pytest, pytest-asyncio, Ruff, Ty, Hatchling, GitHub Actions

# typed_store/specs.py
from typed_store.errors import BulkQueryShapeError

def assert_bulk_compatible(self) -> None:
    invalid_parts: list[str] = []
    if self.order_by:
        invalid_parts.append("order_by")
    if self.limit is not None:
        invalid_parts.append("limit")
    if self.offset is not None:
        invalid_parts.append("offset")
    if self.options:
        invalid_parts.append("options")
    if invalid_parts:
        joined = ", ".join(invalid_parts)
        raise BulkQueryShapeError(
            f"Bulk mutation only supports filter-only Query objects; got unsupported fields: {joined}."
        )
```

# typed_store/sync.py
from sqlalchemy import Engine, delete as sa_delete, func, select, update as sa_update

def bulk_update(
    self,
    model: type[TModel],
    *,
    query: Query[TModel],
    patch: Patch[TModel],
    session: Session | None = None,
    commit: bool = True,
) -> int:
    query.assert_bulk_compatible()
    stmt = sa_update(model).values(**dict(patch.values))
    if query.filters:
        stmt = stmt.where(*query.filters)
    if session is not None:
        result = session.execute(stmt)
        if commit:
            session.commit()
        return int(result.rowcount or 0)

    with self.provider.get_session() as managed_session:
        result = managed_session.execute(stmt)
        if commit:
            managed_session.commit()
        return int(result.rowcount or 0)


def bulk_delete(
    self,
    model: type[TModel],
    *,
    query: Query[TModel],
    session: Session | None = None,
    commit: bool = True,

# examples/bulk_operations.py
from __future__ import annotations

from sqlalchemy.orm import DeclarativeBase, Mapped, mapped_column

from typed_store import Patch, Query, SyncTypedStore, TypedStoreModel


class Base(DeclarativeBase):
    pass


class User(Base, TypedStoreModel):
    __tablename__ = "users"

    id: Mapped[int] = mapped_column(primary_key=True, autoincrement=True)
    name: Mapped[str]
    role: Mapped[str]


store = SyncTypedStore.from_url("sqlite:///bulk_operations.sqlite3")
Base.metadata.create_all(store.engine)

users = User.bind(store)
users.insert_many(
    [
        User(name="alice", role="member"),
        User(name="bob", role="member"),
        User(name="eve", role="guest"),
    ]
)

updated = users.bulk_update(
    query=Query[User]().where(User.role == "member"),
    patch=Patch[User]({"role": "staff"}),
)
deleted = users.bulk_delete(query=Query[User]().where(User.role == "guest"))

print(f"bulk updated: {updated}")
print(f"bulk deleted: {deleted}")
```
